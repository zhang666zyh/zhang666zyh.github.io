Step02. Java数据流

  • InputStream

    字节流

  • InputStreamReader

    字节流通向字符流的桥梁,它读取字节,并使用指定的charset将其解码为字符

    InputStreamReader不是InputStream的子类

    其实就是将字节流转换为字符流

    为了最大的效率,请考虑在BufferedReader中包装一个InputStreamReader,例如：

      BufferedReader in = new BufferedReader(new InputStreamReader(System.in));

  • BufferedReader

    从字符输入流读取文本,缓冲字符,以提供字符,数组和行的高效读取

    可以指定缓冲区大小,或者可以使用默认大小,默认值足够大,可用于大多数用途

    通常,由读取器做出的每个读取请求将引起对底层字符或字节流的相应读取请求.因此,建议将BufferedReader包装在其read()操作可能昂贵的读取器上

    例如FileReaders和InputStreamReaders

    例如

      BufferedReader in = new BufferedReader(new FileReader("foo.in"));

import java.util.Properties;

public class jdbcUtil {
  
  static Properties properties = null;
  
  static {
    properties = new Properties();
    try {
      properties.load(new FileInputStream(new File("E:\\Code\\代码\\Java\\Eclipse-Java01\\MySQL\\lib\\db.properties")));
    } catch (FileNotFoundException e) {
      // TODO Auto-generated catch block
      e.printStackTrace();
    } catch (IOException e) {
      // TODO Auto-generated catch block
      e.printStackTrace();
    }
  }
  
  public static Connection getMySQLConnection() {
    try {
      Class.forName(properties.getProperty("MySQLDriver"));
      return DriverManager.getConnection(properties.getProperty("MySQLURL"),properties.getProperty("MySQLUname"),properties.getProperty("MySQLPwd"));
    } catch (Exception e) {
      // TODO Auto-generated catch block
      e.printStackTrace();
      return null;
    }
  }
  
  public static void Closes(PreparedStatement ps,Connection conn) {
    if(ps != null) {
      try {
        ps.close();
      } catch (SQLException e) {
        // TODO Auto-generated catch block
        e.printStackTrace();
      }
    }
    if(conn != null) {
      try {
        conn.close();
      } catch (SQLException e) {
        // TODO Auto-generated catch block
        e.printStackTrace();
      }
    }
  }
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.util.StringTokenizer;

public class MyFrame extends JFrame implements ActionListener {
    TextField text1,text2;

    public MyFrame(){
        //设置流式布局,FlowLayout是JavaSwing中最简单的布局方式
        setLayout(new FlowLayout(FlowLayout.LEFT,150,20));

        setTitle("实时计算器");
        setSize(800,800);
        setVisible(true);

        text1 = new TextField(20);
        text2 = new TextField(20);
        add(text1);
        add(text2);
        text1.addActionListener((ActionListener) this);
        setVisible(true);
        validate();
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        String inputNumber = e.getActionCommand();
        StringTokenizer stringTokenizer = new StringTokenizer(inputNumber,",'\n'");
        int n = stringTokenizer.countTokens();
        String[] array = new String[n];

        for(int i = 0;i < n;i++){
            array[i] = stringTokenizer.nextToken();
        }

        int sum = Integer.parseInt(array[0]) + Integer.parseInt(array[1]);
        int average = sum / 2;
        text2.setText("和为:" + sum + "," + "平均值为:" + average);
    }
}

                array[1] = e.getActionCommand();

    public void actionPerformed(ActionEvent e) {
        if(e.getSource() == jButton1){
            jTextField3.setText(Integer.parseInt(array[0]) + Integer.parseInt(array[1])+ "");
        }else if(e.getSource() == jButton2){
            jTextField3.setText(Integer.parseInt(array[0]) - Integer.parseInt(array[1])+ "");

            jTextField3.setText(Integer.parseInt(array[0]) * Integer.parseInt(array[1])+ "");

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.*;

public class NotePad {
    private JFrame jFrame;

    private MenuBar menuBar;
    private Menu fileMenu,saveMenu;
    private MenuItem openItem,saveItem,closeItem;

    private JTextArea jtextArea;

    private FileDialog openDia,saveDia;

    private File file;

    public NotePad(){
        init();
    }

    public void init(){
        jFrame = new JFrame("简易记事本");

        menuBar = new MenuBar();
        fileMenu = new Menu("File(f)");
        openItem = new MenuItem("open(o)");
        saveItem = new MenuItem("save(s)");
        closeItem = new MenuItem("exit(e)");

        jtextArea = new JTextArea();
        jFrame.add(jtextArea);

        fileMenu.add(openItem);
        fileMenu.add(saveItem);
        fileMenu.add(closeItem);
        menuBar.add(fileMenu);
        jFrame.setMenuBar(menuBar);

        openDia = new FileDialog(jFrame,"打开文件",FileDialog.LOAD);
        saveDia = new FileDialog(jFrame,"保存为",FileDialog.SAVE);

        jFrame.setBounds(500,500,600,800);
        jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        jFrame.setVisible(true);

        jFrame.validate();

        event();
    }

    public void event(){
        openItem.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                try {
                    openFile();
                } catch (IOException ioException) {
                    ioException.printStackTrace();
                }
            }
        });

        saveItem.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                saveFile();
            }
        });

        closeItem.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                System.exit(0);
            }
        });
    }

    public void openFile() throws IOException {
        openDia.setVisible(true);

        String filePath = openDia.getDirectory();
        String fileName = openDia.getFile();

        if(filePath == null || fileName == null){
            return;
        }

        jtextArea.setText("");

        file = new File(filePath,fileName);
        try {
            BufferedReader bufferedReader = new BufferedReader(new FileReader(file));
            String content = null;
            while((content = bufferedReader.readLine()) != null){
                jtextArea.append(content + "\r\n");
            }
            bufferedReader.close();
        } catch (FileNotFoundException e) {
            e.printStackTrace();
        }
    }

    public void saveFile(){
        if(file == null){
            saveDia.setVisible(true);

            String filePath = saveDia.getDirectory();
            String fileName = saveDia.getFile();

            if(filePath == null || fileName == null){
                return;
            }

            file = new File(filePath,fileName);
        }

        try {
            BufferedWriter bufferedWriter = new BufferedWriter(new FileWriter(file));

            String content = jtextArea.getText();

            bufferedWriter.write(content);
            bufferedWriter.flush();
            bufferedWriter.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

  • 缺点是,synchronized一旦锁定资源,一个线程运行时其他线程只能眼巴巴地等着

  • 这样很影响效率

  • 因此就需要一种机制可以不让一直等待的线程无限等待下去,使用Lock就可以解决这个问题

Lock

  • Lock类使得如果多个线程只是进行读操作,线程之间不会产生冲突

  • Lock还可以知道线程有没有成功获取到锁,这是synchronized无法办到的

  • 所以,Lock提供了比synchronized更多的功能

  • ReentrantReadWriteLock实现了ReadWriteLock接口,ReadWriteLock只有两个方法,读锁和写锁

    package com.zhang.Lock;
  
import java.util.concurrent.locks.ReentrantReadWriteLock;
  
/**
 * 使用读写锁，可以实现读写分离锁定，读操作并发进行，写操作锁定单个线程
 * 
 * 如果有一个线程已经占用了读锁，则此时其他线程如果要申请写锁，则申请写锁的线程会一直等待释放读锁。
 * 如果有一个线程已经占用了写锁，则此时其他线程如果申请写锁或者读锁，则申请的线程会一直等待释放写锁。
 * @author
 *
 */
public class MyReentrantReadWriteLock {
   private ReentrantReadWriteLock rwl = new ReentrantReadWriteLock();
       
      public static void main(String[] args)  {
          final MyReentrantReadWriteLock test = new MyReentrantReadWriteLock();
  	         
          new Thread(){
              public void run() {
                  test.get(Thread.currentThread());
                  test.write(Thread.currentThread());
              };
          }.start();
  	         
          new Thread(){
              public void run() {
                  test.get(Thread.currentThread());
                  test.write(Thread.currentThread());
              };
          }.start();
  	         
      }  
  	    
      /**
       * 读操作,用读锁来锁定
       * @param thread
       */
      public void get(Thread thread) {
          rwl.readLock().lock();
          try {
              long start = System.currentTimeMillis();
  	             
              while(System.currentTimeMillis() - start <= 1) {
                  System.out.println(thread.getName()+"正在进行读操作");
              }
              System.out.println(thread.getName()+"读操作完毕");
          } finally {
              rwl.readLock().unlock();
          }
      }
   
      /**
       * 写操作，用写锁来锁定
       * @param thread
       */
      public void write(Thread thread) {
          rwl.writeLock().lock();;
          try {
              long start = System.currentTimeMillis();
  	             
              while(System.currentTimeMillis() - start <= 1) {
                  System.out.println(thread.getName()+"正在进行写操作");
              }
              System.out.println(thread.getName()+"写操作完毕");
          } finally {
              rwl.writeLock().unlock();
          }
      }
}

package com.first;

import java.io.File;
import java.io.IOException;

public class FirstIO {
	public static void main(String[] args) {
		createFile();
	}
	
	public static void createFile() {
		/*
		 * System.getProperty("user.dir"); 获取当前项目路径
		 * File file = new File(path); 创建文件对象
		 * file.createNewFile(); 创建文件
		 */
		File file = new File(System.getProperty("user.dir") + "/static/file/File/first.txt");
		File file1 = new File(System.getProperty("user.dir") + "/static/file/File");
		try {
			System.out.println(file.createNewFile()); // 新建文件,文件夹不存在会异常,文件已经存在返回false
			System.out.println(file.exists()); // 判断目标是否存在
			System.out.println(file.mkdir()); // 创建new File参数中不存在的单层文件夹
			System.out.println(file.mkdirs()); // 创建new File参数中不存在的多层文件夹
			System.out.println(file.isFile()); // 判断目标是否为文件
			System.out.println(file.isDirectory()); // 判断目标是否为文件夹
			System.out.println(file.getName()); // 获取文件名
			System.out.println(file.getParent()); // 获取文件的父路径
			System.out.println(file.getAbsolutePath()); // 获取文件的绝对路径
//			System.out.println(file.delete()); // 删除自身文件和空文件夹
			
			System.out.println("------------------------------------");
			System.out.println(file1.list()[0] + " " + file1.list()[1]); // 获取当前目录下的所有文件名,结果为List[]	
			
			// 返回当前目录下所有文件的绝对路径,结果为File[]
			System.out.println(file1.listFiles()[0]);
			System.out.println(file1.listFiles()[1]);
		} catch (IOException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
	}
}

package inputStreamIo;

import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;

public class InputStreamIoByte {
	public static void main(String[] args) throws IOException {
		inputStream();
	}
	
	public static void inputStream() throws IOException {
		byte[] buffer = new byte[1024]; 
		int numberRead = 0;
		InputStream inputStream = null;
		OutputStream outputStream = null;
		
		try {
			inputStream = new FileInputStream(System.getProperty("user.dir") + "/static/file/InputStream/InputStreamByte/byte.txt");
			outputStream = new FileOutputStream(System.getProperty("user.dir") + "/static/file/InputStream/InputStreamByte/byte_copy.txt");
			
			// numberRead的作用是防止最后一次读取字节长度小于buffer
			while((numberRead = inputStream.read(buffer)) != -1) {
				outputStream.write(buffer,0,numberRead);
			}
		} catch (FileNotFoundException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
	}
}
