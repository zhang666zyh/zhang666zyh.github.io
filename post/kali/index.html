    # 移动压缩包至opt目录
mv jdk-11.0.5_linux-x64_bin.tar.gz /opt
   
#切换至opt目录进行解压
cd /opt
   
#解压
tar -zxvf jdk-11.0.5_linux-x64_bin.tar.gz
   
#拷贝解压后Java文件夹至/usr/bin
cp -r jdk-11.0.5/ /usr/bin
   
#安装并注册
update-alternatives --install /usr/bin/java java /opt/jdk-11.0.5/bin/java 1
update-alternatives --install /usr/bin/javac javac /opt/jdk-11.0.5/bin/javac 1
update-alternatives --set java /opt/jdk-11.0.5/bin/java
update-alternatives --set javac /opt/jdk-11.0.5/bin/javac
   
#检查是否成功配置Java环境
java -version
