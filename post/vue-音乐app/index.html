    import { createRouter, createWebHashHistory } from 'vue-router'
  
const routes = [{
    path: '/',
    name: 'index',
    component: () =>
        import ("@/views/Index.vue"),
    redirect: 'recommend', // redirect是访问上面的path的路由时,重定向到redirect所代表的路由,一般都是用来设置默认显示的tab选项,将哪个子路由的页面渲染到页面上,这样才能将底下的子组件渲染到页面上
    children: [{
        path: '/recommend',
        name: 'recommend',
        component: () =>
            import ("@/views/Recommend.vue"),
    }, {
        path: '/singer',
        name: 'singer',
        component: () =>
            import ("@/views/Singer.vue"),
    }, {
        path: '/toplist',
        name: 'toplist',
        component: () =>
            import ("@/views/Toplist.vue"),
    }, {
        path: '/search',
        name: 'search',
        component: () =>
            import ("@/views/Search.vue"),
    }]
}, ]
  
const router = createRouter({
    history: createWebHashHistory(),
    routes
})
  
export default router

  • Index.vue的子路由所对应的vue组件

  • 会在点击Index.vue中的切换路由组件时,被渲染到Index.vue的<router-view/>区域

  • 一个页面可以有多个<router-view/>,但你要声明这些<router-view/>的name属性,并和子路由一一对应

  • 就是说,你点击页面的推荐,Recommend.vue,就会被渲染到页面上的<router-view/>

  • 点击页面的搜索,Search.vue,就会被渲染到页面上的<router-view/>
