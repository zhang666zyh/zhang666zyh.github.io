一个对象,原来不是空,当你把它赋值为 null,此时它原来对应的堆内存中的数据就没人使用了

对于GC来说,这就是垃圾

  • 当然,JavaScript有自动的垃圾回收机制,平时不需要开发者操作

  • 对过程进行一个讲解

    对象是引用类型,new一个对象时,变量名和值都存储在栈内存中,因为值只是一个地址,这个地址指向堆内存中的数据

    当将对象赋值为null时,实质上是变量的栈内存中的地址为null,而不是堆内存中的数据为null

    此时堆内存中的数据依然存在却没人使用

    GC就把它当作垃圾

    JavaScript引擎有GC机制,自动垃圾回收机制

<script type="text/javascript">

    /* 也可以通过 /x/y 的模式进行匹配,第一个参数是正则表达式,第二个是匹配规范 */
    var regExp2 = /ab/i;
    console.log(regExp2.test("abcsafa"));

    /* |是或的意思,只要匹配串中有这三个任意一个都匹配成功 */
    var regExp3 = /a|b|c/i;
    console.log(regExp3.test("fahahfanffabav"));

    /*
     * [] == |
     * a|b和[ab]是等价的
     */
    var regExp4 = /[ab]/i;
    console.log(regExp4.test("fajhdadfauroewh"));

    /*
     * [a-z]会匹配所有的小写字母
     * [A-Z]会匹配所有的大写字母
     * [A-z]表示任意字母
     */
    var regExp5 = /[a-z]/i;
    console.log(regExp5.test("fha82u4roy2342y4@4"));

    /* 匹配abc、aec或是agc,可以这样简化写 */
    var regExp6 = /a[beg]c/i;
    console.log(regExp6.test("fahaecjfalfhao"));

    /* ^是除了的意思,除了小写字母之外的全部内容 */
    var regExp7 = /[^a-z]/i;
    console.log(regExp7.test("ru03r82na@$@afa"));

    /* 量词,{n}代表字符要出现n次,它只对前面的第一个内容起作用,如果不想只匹配前面一个字符而需要匹配前面多个字符串出现多次,使用括号将需要匹配的字符括起来 */
    var regExp8 = /a{1}/g;
    console.log("82y4rfnaafaaf".match(regExp8));

    /* {m,n}代表前面第一个内容出现m到n次 */
    var regExp9 = /a{1,5}/g;
    console.log("824oaf;a83ry8rafafa".match(regExp9));

    /* {n,}代表至少出现n次 */
    var regExp10 = /a{3,}/g;
    console.log("afu84y204-08 a'".match(regExp10));

    /* + 代表一次或者多次 */
    var regExp11 = /a+/g;
    console.log("aaaaaaaaaaaaaaaaafafafafhgu8u0aaaa".match(regExp11));

    /* * 代表0次或者多次 */
    var regExp12 = /a*/g;
    console.log("aaaaaaaaaaaaaaaaafafafafhgu8u0aaaa".match(regExp12));

    /* ? 代表0次或1次 */
    var regExp13 = /a?/g;
    console.log("aaaaaaaaaaaaaaaaafafafafhgu8u0aaaa".match(regExp13));

    /* ^代表以...开头 */
    var regExp14 = /^he/g;
    console.log(regExp14.test("helloworld"));  

    /* $代表以...结尾 */
    var regExp15 = /ld?/g;
    console.log(regExp15.test("helloworld"));

    /* 如果同时使用^和$进行匹配,那么整个字符串必须完全匹配正则 */
    var regExp16 = /^he.*ld$/;

    var regExp17 = /./g;
