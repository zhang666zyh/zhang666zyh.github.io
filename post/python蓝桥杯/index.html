# 折纸
# A0是1189*841,从A0~A9分别是A0沿长边对折后的尺寸,输入一个尺寸(A0~A9),输出纸张的长宽

# 做法1,最简单的思维
def one():
    n = input()
    index = ["A0", "A1", "A2", "A3", "A4", "A5", "A6", "A7", "A8", "A9"]
    row = []
    column = []
    left = 1189
    right = 841

    for i in range(len(index)):
        a = max(left, right)
        b = min(left, right)
        row.append(a)
        column.append(b)
        left, right = a // 2, b  # 让下一次的长宽保持为最大和最小,且最大的要折半取整

    if n in index:
        print(row[index.index(n)])  # 列表通过索引取值 -> 列表名.index(索引)
        print(column[index.index(n)])


# 做法2,牛逼
def two():
    n = int((list(input()))[1])
    left, right = 1189, 841

    for i in range(0, n):
        left = left // 2
        if left < right: left, right = right, left

    print(left)
    print(right)

# 将1-n的所有数字,按照 每一位的独立数字的加和 排序,这样的排序方法,第m个数是多少
n = int(input())
m = int(input())

value = [i for i in range(1, n + 1)]  # [i for i in range(a,b)]的方式,将每次循环的i赋值给最前面的变量i,并使用append的方式插入到列表中
index = []

for item in value:
    index.append(sum(list(map(int, list(str(item))))))  # map(type,list),将list中所有元素批量转换数据类型, sum,对可迭代对象中元素自动求和

'''
    a = (1,2,3)
    b = (4,5,6)
    list(zip(a,b)) = [(1,4),(2,5),(3,6)]
    zip的结果是个可迭代对象,所以必须使用list转换一下
'''
# sorted函数排序的对象如果是键值对,有个参数 reverse,True就是倒序 == 如果是 (1,2) , (3,4) , (5,6)这种也是一样,默认按照第一个元素(键)排序
res = [i for _, i in sorted(zip(index, value))]
print(res[m - 1])
