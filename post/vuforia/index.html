using UnityEngine;
using Vuforia;
  
/// <summary>
/// A custom handler that implements the ITrackableEventHandler interface.
///
/// Changes made to this file could be overwritten when upgrading the Vuforia version.
/// When implementing custom event handler behavior, consider inheriting from this class instead.
/// </summary>
public class MyDefaultTrackableEventHandler : MonoBehaviour, ITrackableEventHandler
{
    // 人物模型
    public GameObject aiXiPrefab;
      
    #region PROTECTED_MEMBER_VARIABLES
  
    protected TrackableBehaviour mTrackableBehaviour;
    protected TrackableBehaviour.Status m_PreviousStatus;
    protected TrackableBehaviour.Status m_NewStatus;
  
    #endregion // PROTECTED_MEMBER_VARIABLES
  
    #region UNITY_MONOBEHAVIOUR_METHODS
  
    protected virtual void Start()
    {
        mTrackableBehaviour = GetComponent<TrackableBehaviour>();
        if (mTrackableBehaviour)
            mTrackableBehaviour.RegisterTrackableEventHandler(this);
    }
  
    protected virtual void OnDestroy()
    {
        if (mTrackableBehaviour)
            mTrackableBehaviour.UnregisterTrackableEventHandler(this);
    }
  
    #endregion // UNITY_MONOBEHAVIOUR_METHODS
  
    #region PUBLIC_METHODS
  
    /// <summary>
    ///     Implementation of the ITrackableEventHandler function called when the
    ///     tracking state changes.
    /// </summary>
    public void OnTrackableStateChanged(
        TrackableBehaviour.Status previousStatus,
        TrackableBehaviour.Status newStatus)
    {
        m_PreviousStatus = previousStatus;
        m_NewStatus = newStatus;
          
        Debug.Log("Trackable " + mTrackableBehaviour.TrackableName + 
                  " " + mTrackableBehaviour.CurrentStatus +
                  " -- " + mTrackableBehaviour.CurrentStatusInfo);
  
        if (newStatus == TrackableBehaviour.Status.DETECTED ||
            newStatus == TrackableBehaviour.Status.TRACKED ||
            newStatus == TrackableBehaviour.Status.EXTENDED_TRACKED)
        {
            OnTrackingFound();
        }
        else if (previousStatus == TrackableBehaviour.Status.TRACKED &&
                 newStatus == TrackableBehaviour.Status.NO_POSE)
        {
            OnTrackingLost();
        }
        else
        {
            // For combo of previousStatus=UNKNOWN + newStatus=UNKNOWN|NOT_FOUND
            // Vuforia is starting, but tracking has not been lost or found yet
            // Call OnTrackingLost() to hide the augmentations
            OnTrackingLost();
        }
    }
  
    #endregion // PUBLIC_METHODS
  
    #region PROTECTED_METHODS
  
    // 找到人物模型
    protected virtual void OnTrackingFound()
    {
        // 生成人物模型
        GameObject aiXi = GameObject.Instantiate(aiXiPrefab);
  
        // 将模型的位置设置成卡片的位置
        aiXi.transform.position = this.transform.position;
  
        // 将模型设置成卡片的子元素
        aiXi.transform.parent = this.transform;
    }
  
  
    protected virtual void OnTrackingLost()
    {
        Destroy(GameObject.Find("aiXi(Clone)"));
    }
  
    #endregion // PROTECTED_METHODS
}

  • 原理就是当发现卡片时候展示模型,同时将模型位置固定到卡片位置

  • 当卡片消失后,将模型销毁,销毁之前的动画也随之消失了

  • 然后将ImageTarget的DefaultTrackableEventHandler.cs脚本删掉,将自己的MyDefaultTrackableEventHandler.cs赋给ImageTarget

  • 将人物模型预制体赋给ImageTarget

    using System.Collections;
using System.Collections.Generic;
using UnityEngine;
  
public class TouchTap : MonoBehaviour
{
    // Start is called before the first frame update
    void Start()
    {
          
    }
  
    // Update is called once per frame
    void Update()
    {
        // 如果点击了屏幕
        if(Input.GetMouseButtonDown(0)){
            // 从当前脚本所从属的摄像机发射一条射线到参数的位置,这里是点击的位置
            Ray ray = Camera.main.ScreenPointToRay(Input.mousePosition);
            // RaycastHit是用来接收射线反射的信息的
            RaycastHit hitInfo;
  
            // 如果照射到了collider,则将反射的信息输出到hitInfo
            if(Physics.Raycast(ray,out hitInfo)){
                /*
                 * Input.GetTouch(index);index用于指定什么设备触摸屏幕--->0是手指...
                 */
                 /*
                  * TouchPhase用于描述触摸状态
                    * TouchPhase.Began ---> 一根手指触摸到了屏幕
                    * TouchPhase.Canceled
                    * TouchPhase.Ended
                    * TouchPhase.Moved
                    * TouchPhase.Stationary
                  */
                // 如果触摸一次屏幕 且 一根手指触摸到了屏幕
                // 注意:touchCount是触摸,tapCount是点击
                if(Input.touchCount == 1 && Input.GetTouch(0).phase == TouchPhase.Began){
                    // 点击数量
                    if(Input.GetTouch(0).tapCount == 2){
                        // 摧毁光线照射到的collider
                        Destroy(hitInfo.collider.gameObject);
                    }
                }
            }
        }
    }
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class TouchTap : MonoBehaviour
{
    private float touchTime;
    private bool newTouch = false;

    // Start is called before the first frame update
    void Start()
    {
        
    }

    // Update is called once per frame
    void Update()
    {
        /*
         * Input.GetMouseButton ---> 当前鼠标点是否被按住
         * Input.GetMouseButtonUp ---> 当前鼠标点是否被按下过(点击一下/点击了)
         * Input.GetMouseButtonDown ---> 当前鼠标点是否被抬起
         * 索引依然是0代表手指,1、2、3...代表鼠标、触控笔等
         */
        // 如果点击了一下屏幕
        // if(Input.GetMouseButtonDown(0)){
        // 返回给定的鼠标点位置是否被按住
        if(Input.GetMouseButton(0)){
            // 从当前脚本所从属的摄像机发射一条射线到参数的位置,这里是点击的位置
            Ray ray = Camera.main.ScreenPointToRay(Input.mousePosition);
            // RaycastHit是用来接收射线反射的信息的
            RaycastHit hitInfo;

            // 如果照射到了collider,则将反射的信息输出到hitInfo
            if(Physics.Raycast(ray,out hitInfo)){
                /* 双击 */
                /*
                 * Input.GetTouch(index);index用于指定什么设备触摸屏幕--->0是手指...
                 */
                 /*
                  * TouchPhase用于描述触摸状态
                    * TouchPhase.Began ---> 一根手指触摸到了屏幕
                    * TouchPhase.Canceled
                    * TouchPhase.Ended
                    * TouchPhase.Moved
                    * TouchPhase.Stationary ---> 触摸屏幕后不松开(长按)
                  */
                // 如果触摸一次屏幕 且 一根手指触摸到了屏幕
                // 注意:touchCount是触摸,tapCount是点击
                // if(Input.touchCount == 1 && Input.GetTouch(0).phase == TouchPhase.Began){
                    // 点击数量
                    // if(Input.GetTouch(0).tapCount == 2){
                        // 摧毁光线照射到的collider
                        // Destroy(hitInfo.collider.gameObject);
                    // }

                /* 长按 */
                /* 如果触摸一次屏幕 */
                if(Input.touchCount == 1){
                    /* 获取点击状态 */
                    Touch touch = Input.GetTouch(0);

                    /* 如果一个手指触摸屏幕*/
                    if(touch.phase == TouchPhase.Began){
                        newTouch = true;
                        /* 获取当前系统时间 */
                        touchTime = Time.time;
                    /* 触摸屏幕后不松开 */
                    }else if(touch.phase == TouchPhase.Stationary){
                        /* 如果长按时间大于1s*/
                        if(newTouch == true && Time.time - touchTime > 1f){
                            newTouch = false;
                            Destroy(hitInfo.collider.gameObject);
                        }
                    // 只有一根手指点击了屏幕后还得长按才销毁,否则都是false
                    // 这样直到手指离开屏幕进行下一次的触摸点击才跳出本次失败的循环进入下一次
                    }else{
                        newTouch = false;
                    }
                }
            }
        }
    }
}

	}

                // 如果触摸状态是移动
                if(Input.GetTouch(0).phase == TouchPhase.Moved){
                    // 绕着物体自身x轴旋转
                    transform.Rotate(Vector3.up * Input.GetAxis("Mouse X") * -speed * Time.deltaTime);
                }
            }
        }
    }
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Enlarge : MonoBehaviour
{
    // 物体的两个旧位置点
    private Vector2 oldP1;
    private Vector2 oldP2;

    // Start is called before the first frame update
    void Start()
    {
        
    }

    // Update is called once per frame
    void Update()
    {
        // 如果两根手指触碰了屏幕
        if(Input.touchCount == 2){
            // 如果两根手指移动或者一根手指移动一根手指静止
            if(Input.GetTouch(0).phase == TouchPhase.Moved || Input.GetTouch(1).phase == TouchPhase.Moved){
                // 获取当前触摸点手指位置
                Vector2 tempP1 = Input.GetTouch(0).position;
                Vector2 tempP2 = Input.GetTouch(1).position;

                // 将获取的手指触摸点位置作为新位置,和旧位置一起传入判断放大还是缩小的方法中
                // 放大
                if(isEnlarge(oldP1,oldP2,tempP1,tempP2)){
                    // transform.localScale可以获取物体x、y、z三个方向的长度
                    float oldScale = transform.localScale.x;
                    // 在物体原有长度的基础上增长1.025倍
                    float newScale = oldScale * 1.025f;

                    // 放大物体
                    transform.localScale = new Vector3(newScale,newScale,newScale);
                }else{
                    float oldScale = transform.localScale.x;
                    float newScale = oldScale / 1.025f;

                    transform.localScale = new Vector3(newScale,newScale,newScale);
                }

                oldP1 = tempP1;
                oldP2 = tempP2;
            }
        }
    }

    bool isEnlarge(Vector2 oldP1,Vector2 oldP2,Vector2 newP1,Vector2 newP2){
        float oldLength = Mathf.Sqrt((oldP1.x - oldP2.x) * (oldP1.x - oldP2.x) + (oldP1.y - oldP2.y) * (oldP1.y - oldP2.y));
        float newLength = Mathf.Sqrt((newP1.x - newP2.x) * (newP1.x - newP2.x) + (newP1.y - newP2.y) * (newP1.y - newP2.y));

        if(oldLength < newLength){
            return true;
        }else{
            return false;
        }
    }
}
