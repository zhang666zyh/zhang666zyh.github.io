Step06.complex

  • go语言的复数,分为complex64和complex128两种类型
  • complex64位的实部和虚部,都为32位数据(整数、浮点数) - 两个32位加起来就是64
  • complex128位的实部和虚部,都为64位数据(整数、浮点数) - 两个64位加起来就是128

package main

import (
	"fmt"
	"strings"
)

func main(){
	c1 := 'c' // go中的单引号,就是字符,双引号就是字符串,不能混用,有严格的区分
	fmt.Printf("%c", c1)

	fmt.Println("\n-------------------------------------")

	s1 := "\"D:\\CodePath\\hugo\"" // 使用 / 对特殊字符进行转义
	fmt.Printf("%v",s1)

	fmt.Println("\n-------------------------------------")

	fmt.Printf("%d", len(s1)) // 计算字符串长度

	fmt.Println("\n-------------------------------------")

	// 字符串拼接
	name := "智博"
	world := "科技"
	res := name + world // 第一种方法,直接将两个字符串相加 
	fmt.Println(res)
	res1 := fmt.Sprintf("%s%s", name, world) // 第二种方法,使用fmt.Sprintf方法,用占位符将传入的变量返回为字符串，不会在终端显示 , 性能上据说很拉垮
	fmt.Printf("%v", res1)

	fmt.Println("\n-------------------------------------")

	cut := strings.Split(s1, "\\") // 将字符串按照指定字符串进行切片
	fmt.Printf("%v", cut)
	fmt.Println("\n-------------------------------------")

	fmt.Printf("%v", strings.Contains(s1, "hugo")) // 判断字符串中,是否存在某个字符串

	fmt.Println("\n-------------------------------------")
	ss := "aassddff"

	fmt.Printf("%v", strings.HasPrefix(ss, "aa")) // 判断字符串a的前缀是否为b
	fmt.Println()
	fmt.Printf("%v", strings.HasSuffix(ss, "aa")) // 判断字符串a的后缀是否为b

	fmt.Println("\n-------------------------------------")

	fmt.Printf("%v", strings.Index(ss, "s")) // 判断字符串a中,是否存在字符串b。如果存在,返回b在a中第一次出现的索引;如果不存在返回-1
	fmt.Printf("%v", strings.LastIndex(ss, "d")) // 判断字符串a中,是否存在字符串b。如果存在,返回b在a中最后一次出现的索引;如果不存在返回-1

	fmt.Println("\n-------------------------------------")
	fmt.Printf("%v", strings.Join(cut, "-")) // 将数组中元素,用指定字符串作为插值串联起来 , 例如: [“a“, “b“, “c“] “-“ --->  "a-b-c"

}

import "fmt" 

func main(){
	for i:=0; i<10; i++{
		fmt.Println(i)
	}

	fmt.Println("\n--------------")

	// for循环里没声明变量,就往上找
	j := 2
	for ; j < 10; j++{
		fmt.Println(j)
	}

	fmt.Println("\n--------------")

	var k = 5
	for k < 10{ // 还是往上找,如果for循环的条件一直成立,循环就会一直执行下去
		fmt.Println(k)
		k++
	}

	fmt.Println("\n--------------")

	// 无限循环
	// for {
	// 	fmt.Println("111")
	// }

	fmt.Println("\n--------------")

	// range,遍历集合元素的索引和值
	s := "hello张"
	for index,value := range s{
		fmt.Printf("str[%v] = %v\n", index,value)
	}

package main

import "fmt"

func main() {
	var n = 5

	// go的switch-case,执行完一个case不会继续执行下一个case,自动break
	switch n {
	case 1:
		fmt.Println("大拇指!!!")
	case 2:
		fmt.Println("食指!!!")
	case 3:
		fmt.Println("中指!!!")
	case 4:
		fmt.Println("第四个指头!!!")
	case 5:
		fmt.Println("小拇指!!!")
	default:
		fmt.Println("六指琴魔!!!")
	}

	switch m := 6; m {
	case 1:
		fmt.Println("大拇指!!!")
	case 2:
		fmt.Println("食指!!!")
	case 3:
		fmt.Println("中指!!!")
	case 4:
		fmt.Println("第四个指头!!!")
	case 5:
		fmt.Println("小拇指!!!")
	default:
		fmt.Println("六指琴魔!!!")
	}

	switch num := 7; num {
	case 1, 3, 5, 7, 9:
		fmt.Println("奇数")
	case 2, 4, 6, 8, 10:
		fmt.Println("偶数")
	default:
		fmt.Println("错误")
	}

	// switch中的case中也可以写判断语句,这样,switch后就不需要写判断变量和声明变量了
	judge := 3
	switch {
	case judge > 0:
		fmt.Println("大于0")
	case judge > 2:
		fmt.Println("大于2")
	}
}

package main

import "fmt"

func main() {
	// 使用make函数构造切片,初始化都为0
	a := make([]int,2,10)  
	fmt.Printf("%v - %v\n", len(a), cap(a))  // len:元素个数 cap:容量
	// 容量是分配给容器的空间,长度是使用的元素个数


	// 切片的引用
	s3 := []int{1,2,3,4,5}
	s4 := s3  // s4引用的是s3,使用的都是那一块连续内存空间
	s4[0] = 100 // 所以更改s4的值,s3也会发生改变
	fmt.Printf("%v-%v\n",s3,s4)

	// 切片扩容
	s5 := []int{1,2,3,4,5}
	fmt.Println(s5)
	s5 = append(s5, 6) // append函数扩容只能使用原来的变量接收返回值,引用类型不可以
	fmt.Println(s5)

	// 切片扩容的策略 - append追加元素,如果原来的底层数组放不下了,Go语言就会把底层数组换一个 - 重新申请一个底层数组(cap)
	/* 
	 *  首先判断，如果新申请容量(cap )大于2倍的旧容量(old.cap ),最终容量(newcap )就是新申请的容量(cap)。
	 *	否则判断，如果旧切片的长度小于1。24 ,则最终容量(newcap)就是旧容量(old.cap)的两倍，即(newcap=doublecap ),
	 *	否则判断,如果旧切片长度大于等于1024，则最终容量(newcap )从旧容量(old.cap )开始循环增加原来的1/4 ,即(ne^vcap=old.cap,for {ne^vcap += newcap/4))直到最终容量(newcap )大于等于新申请的容量(cap), 即(newcap >= cap )
	 *	如果最终容量(cap )计算值溢出，则最终容量(cap )就是新申请容量(cap )
	 * 
	 */
	s6 := make([]int,2,5)
	for i:=0;i < len(s6);i++{ // 这里必须使用len,如果直接使用cap,对指定索引赋值会越界,扩容只能通过append,将值塞入cap中,而不能直接通过索引插入值
		s6[i] = i
	}

	fmt.Printf("s6 - len = %v,cap = %v\n", len(s6),cap(s6))

	s6 = append(s6, 1)
	s6 = append(s6, 2)
	s6 = append(s6, 3)
	s6 = append(s6, 4)
	s6 = append(s6, 5)
	fmt.Printf("s6 - len = %v,cap = %v\n", len(s6),cap(s6))


	// 使用copy函数复制切片
	s7 := []int{1,2,3,4,5}
	var s8 = make([]int,2,5) // 复制的,得声明一个变量接收复制
	copy(s8,s7)  // 只复制len长度,不复制cap长度
	fmt.Println(s8)

	// 删除切片中的内容
	s10 := []int{0,1,2,3,4,5}
	fmt.Println(s10)
	fmt.Printf("%v-%v\n",len(s10),cap(s10))
	s10 = append(s10[:2],s10[3:]...) // 通过将需要的部分切片出来,然后append到原数组,实现删除 修改了底层数组
	fmt.Println(s10)
	fmt.Printf("%v-%v\n",len(s10),cap(s10))
}
  • 切片的本质,就是将数组这一块连续的内存空间,框起来,进行保护
  • 只有引用的和自身可以使用

make和new的区别

  • make和new都可以用来申请内存

  • new一般都是给基本类型申请内存,返回的是这种类型的指针类型(例如你new(int),返回的就是*int),用的少

  • make是用来给slice、map、chan这三种类型申请内存的,返回的是这种类型本身

    1. make(map[string]string)

    2. make([]int, 2)

    3. make([]int, 2, 4)

    第一种用法，即缺少长度的参数，只传类型，这种用法只能用在类型为map或chan的场景，例如make([]int)是会报错的。这样返回的空间长度都是默认为0的。

    第二种用法，指定了长度，例如make([]int, 2)返回的是一个长度为2的slice

    第三种用法，第二参数指定的是切片的长度，第三个参数是用来指定预留的空间长度，例如a := make([]int, 2, 4), 这里值得注意的是返回的切片a的总长度是4，预留的意思并不是另外多出来4的长度，其实是包含了前面2个已经切片的个数的。所以举个例子当你这样用的时候 a := make([]int, 4, 2)，就会报语法错误

	"fmt"
	"math/rand"
	"sort"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano()) // 初始化随机数种子
	var scopeMap = make(map[string]int, 200) // 初始化一个长度为200的map

	// 向map中添加key和value
	for i:=0;i < 200;i++{
		key := fmt.Sprintf("stu%d", i) // fmt.Sprintf,格式化拼接一个字符串并返回
		value := rand.Intn(100) // rand.Intn(n) 返回一个随机的整数 m，且 0 <= m < n
		scopeMap[key] = value 
	}

	// 获取出map中所有的key,存储到数组中
	var keys = make([]string, 0, 200)
	for key := range scopeMap{
		keys = append(keys, key)
	}

	// sort.Strings按照unicode码进行排序, 将map按照key进行排序
	sort.Strings(keys)

	// 遍历刚才提取的map的keys集合
	for _,key := range keys{
		fmt.Println(key, scopeMap[key])
	}
}

Step20.Go 语言切片(Slice)

  • Go 语言切片是对数组的抽象
  • Go 数组的长度不可改变，在特定场景中这样的集合就不太适用
  • Go 中提供了一种灵活，功能强悍的内置类型切片(“动态数组”)
  • 与数组相比切片的长度是不固定的，可以追加元素，在追加时可能使切片的容量增大

声明切片

  • 声明和初始化是两码事,声明的变量,一般都没有被分配内存空间
  • 声明就是创建这么个变量
  • 初始化是给这个变量分配可操作的内存空间
  • make函数,可以在声明的同时,将切片进行初始化
  • 数组中,你需要在长度参数位置,一定要写上...或者数组长度
  • 你可以声明一个未指定大小的数组来定义切片

package main

import (
	"fmt"
)

func fibnacci(n int)(result int){
	if n < 2{
		return n
	}
	return fibnacci(n - 2) + fibnacci(n - 1)
}

	for i:=0;i < 4;i++{
		fmt.Println(fibnacci(i))
	}
}
  • fibnacci数列的算法

  • 并不是说传入一个值,函数将() 这个值作为索引 在数列中的位置 这个位置的值,以及这个值前面一直到0的所有值)一起返回

// 1.声明一个接口
type Phone interface{
    // 接口里有个方法
	call()
}

// 2.声明一个结构体(类),类实现接口
type Iphone struct{

}

// 2.声明一个结构体(类),类实现接口
type TesrraPhone struct{

}

// 3.实现接口 -> func (结构体变量 结构体类型) 接口的某个方法 -> 代表这个结构体"类型""类型""类型",这个"类",实现了这个接口的这个方法
func (iphone Iphone) call(){
	fmt.Println("我是Iphone'类型',我实现了Phone的call方法")

func (tesrraPhone TesrraPhone) call(){
	fmt.Println("我是TesrraPhone'类型',我实现了Phone的call方法")

	var phone Phone

    // 声明一个结构体变量,这个结构体变量实现了接口方法

    // 实现了接口方法的结构体变量,调用实现的方法
	phone.call()

	phone = new(TesrraPhone)
	phone.call()
}

    package main
  
import (
  "fmt"
)
  
// 1. 定义一个 错误信息结构体,作用有两个 , 1.存储除数和被除数 2.继承实现go的error接口,实现error.Error()方法
type DivideError struct{
  dividee int 
  divider int
}
  
// 2. 写个接口,让 错误信息结构体实现error.Error方法,并把信息返回 , 这样这个结构体就变成了错误信息容器,返回的都是错误信息
func (de *DivideError) Error() string{
  strFormat := `
      错误!!!被除数为0
      dividee:%d
      divider:0
  `
  
  return fmt.Sprintf(strFormat, de.dividee)
}
  
// 3. 使用错误处理接口,进行除法运算
func divide(varDividee int, varDivider int) (result int, errorMsg string){
  // 如果除数为0
  if varDivider == 0{
      // 实例化一个结构体对象,属性是被除数和除数
      errorData := DivideError{
          dividee: varDividee,
          divider: varDivider,
      }
  		
      // 调用 结构体实现的接口方法 Error()
      errorMsg = errorData.Error()
      // 退出程序
      return
  // 如果除数不为0
  }else{
      return varDividee / varDivider, ""
  }
}
  
  
func main() {
  fmt.Println(divide(10,0))
}

Step26.并发

  • Go语言最大的特色就是从语言层面支持并发（Goroutine）

  • Goroutine是Go中最基本的执行单元

  • 事实上每一个Go程序至少有一个Goroutine：主Goroutine。当程序启动时，它会自动创建

  • 就是说如果你不声明,一个程序中只有一个goroutine,就都得按顺序排队使用这个goroutine

  • Go 语言支持并发，我们只需要通过 go 关键字来开启 goroutine 即可。

  • goroutine 是轻量级线程，goroutine 的调度是由 Golang 运行时进行管理的。

  • goroutine 语法格式：

    1

    go 函数名( 参数列表 )
    • 例如：
    1

    go f(x, y, z)
    • 开启一个新的 goroutine:
    1

    f(x, y, z)
    • Go 允许使用 go 语句开启一个新的运行期线程， 即 goroutine，以一个不同的、新创建的 goroutine 来执行一个函数。 同一个程序中的所有 goroutine 共享同一个地址空间

    package main
  
import (
  "fmt"
  "time"
)
  
func say(s string){
  for i := 0;i < 5;i++{
      time.Sleep(100 * time.Millisecond)
      fmt.Println(s)
  }
}
  
func main() {
  go say("hello")
  say("world")
}

  • 执行以上代码，你会看到输出的 hello 和 world 是没有固定先后顺序。因为它们是两个 goroutine 在执行
