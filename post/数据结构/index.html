// 使用const代替#define
#define MAXSIZE 20 // 存储空间初始分配量
#define OK 1
#define ERROR 0
#define TRUE 1
#define FALSE 0

typedef int ElemType; // 定义顺序表中元素数据类型
typedef int Status; // 表示状态的int,用于判断返回值是上面宏定义的1还是0
typedef struct{
    ElemType *data;
    int length; // 顺序表当前长度
}SqList;

// C++中main中直接传递结构体变量,型参中就可以直接获取结构体变量的引用地址
Status initList(SqList &L){
    L.data = new ElemType[MAXSIZE]; // new一个顺序表
    // 如果new失败
    if(!L.data){
        // 异常退出程序
        exit(1);
    }

    L.length = 0;
    return OK;
}

// 读操作不需要传递指针
Status displayList(SqList L){
    if(L.length == 0){
        return ERROR;
    }

    int i;
    for(i = 0;i < L.length;i++){
        printf("array[%d]=%d\t",i,L.data[i]);
    }
    printf("-------------------------\n");
}

// 对于形式参数传递结构体指针,获取结构体指针成员变量需要使用 ->
// 如果不是指针是普通类型结构体,使用 .
Status insertList(SqList* L,int i,ElemType e){
    int k;

    // 如果表满退出
    if(L->length == MAXSIZE){
        return ERROR;
    }

    /* 
     *如果插入位置不在范围内
     * 这里的L.length + 1是因为
     *     1.这里指定的是位置不是索引,所以加1,现在最大是L.length
     *     2.因为是插入,允许后插法,所以是L.length + 1
     * 下面插入移动元素时候索引还要减去1,是一样的
     */
    if((i < 1) || (i > L->length + 1)){
        return ERROR;
    }

    // 如果插入位置不是表尾
    if(i <= L->length){
        // 索引是长度 - 1,i - 1是插入位置的索引
        for(k = L->length - 1;k > i - 1;k--){
            L->data[k + 1] = L->data[k];  
        }
    }

    // 将新元素插入
    L->data[i - 1] = e;
    L->length++;
    return OK;
}

// 用e保存(并返回)要删除的数据
Status deleteListElement(SqList* L,int i,ElemType *e){
    int k;

    // 顺序表为空无法删除
    if(L->length == 0){
        return ERROR;
    }

    // 删除位置不在范围内
    if((i < 1) || (i > L->length)){
        return ERROR;
    }

    *e = L->data[i - 1];
    
    // 如果删除的不是表尾元素
    if(i < L->length){
        /*
         * 从删除位置开始是因为 一个元素的位置的数值 正好是要它本身后一个元素的索引
         * 要删除位置的数值 正好是 要删除的元素的后一个元素的索引
         * 直到L->length - 1
         */
        for(k = i;k < L->length;k++){
            // 将要删除的元素后面所有元素前移一位,顺便也把要删除的元素的值覆盖了
            L->data[k - 1] = L->data[k];
        }    
    }
    L->length--;
    return OK;
}

// 通过位置查找,顺序表内部是数组不用遍历,通过位置查找直接数组名[索引]就可以

    // 顺序表为空无需查找
    if(L->length == 0){
        return ERROR;
    }

    // 查找位置不在范围内
    if((i < 1) || (i > L->length)){

    *e = L->data[i - 1];
    printf("第%d个元素值为%d",i,L->data[i - 1]);
    return OK;
}

// 查找指定值,需要遍历
Status findListElementByValue(SqList* L,ElemType value){
    int k;
    bool flag = false;
    
    // 顺序表为空无需查找
    if(L->length == 0){
        return ERROR;
    }

    for(k = 0;k < L->length;k++){
        if(L->data[k] == value){
            flag = true;
            break;
        }
    }

    if(flag){
        printf("找到\n");
        printf("元素%d在%d位置",L->data[k],k + 1);
        return OK;
    }else{
        return ERROR;
    }
}

// 翻转操作
Status reverseList(SqList* L){
    // 顺序表为空无需翻转
    if(L->length == 0){
        return ERROR;
    }

    // 左指针, 右指针, 存放数据的临时变量
    ElemType left = 0;
    ElemType right = L->length - 1;
    ElemType temp = 0;

    while(true){
        // 如果是奇数则左右指针会在表中心元素相遇
        // 如果是偶数则最终左指针会在右指针后面一个位置
        if(left >= right){
            break;
        }

        temp = L->data[left];
        L->data[left] = L->data[right];
        L->data[right] = temp;

        left++;
        right--;
    }
    return OK;
}

// 顺序表的冒泡排序
Status bubbleSort(SqList* L){
    // 顺序表为空无需排序
    if(L->length == 0){
        return ERROR;
    }

    int index = L->length - 1; // 算法外层循环
    bool flag = true; // 用于判断本次循环是否发生交换,如果没发生说明排序完成直接退出,优化了代码,避免了不必要的交换操作
    int i; // for循环的索引
    ElemType temp; // 交换算法存储数据的临时变量

    // 每一次都将最大的放到数组后面,最后剩下第一个肯定是最小的,所以不用等于0
    while((index > 0) || (flag)){
        flag = false;
        for(i = 0;i <= index;i++){
            if(L->data[i] > L->data[i + 1]){
                flag = true; // 发生交换
                temp = L->data[i];
                L->data[i] = L->data[i + 1];
                L->data[i + 1] = temp;
            }
        }
        --index; // 每次循环最大值都被移动到数组相对于所有小值的最后面,所以每次循环后下一次就减少一次比较交换操作
    }
    return OK;
}

Status directInsertSort(SqList* L){
    // 顺序表为空无需排序
    if(L->length == 0){
        return ERROR;
    }

    // 用来存储插入元素的临时变量
    ElemType temp;
    int i,j;

    for(i = 1;i < L->length;i++){
        if(L->data[i] < L->data[i - 1]){
            temp = L->data[i];
            L->data[i] = L->data[i - 1];

            for(j = i - 1;L->data[j] > temp;j--){
                L->data[j + 1] = L->data[j];
            }
            L->data[j + 1] = temp;
        }
    }
    return OK;
}
    // 顺序表的折半插入排序
Status biInsertSort(SqList* L){
    // 顺序表为空无需排序
    if(L->length == 0){
        return ERROR;
    }

    // 用来存储插入元素的临时变量
    ElemType temp;
    int i,j;
    int low,high,mid;

    // 要插入元素的前面的有序数组
    for(i = 1;i < L->length;i++){
        // 将要插入元素保存在临时变量中
        temp = L->data[i];
        low = 0,high = i - 1;

        // 在插入元素前面的有序数组中折半查找插入位置
        while(low <= high){
            mid = (low + high) / 2;
            if(temp < L->data[mid]){
                high = mid - 1;
            }else{
                low = mid + 1;
            }
        }

        // 找到插入位置,这时候(high = low - 1) || (high = low)
        // 应该插入的位置是(high + 1)
        // 总之使用high和low都一样,这里使用high
        // 将从(i - 1)-(要插入元素的前面的有序数组的最后一个元素) 到 (high + 1)-(要插入的位置) 的所有元素向后移动一位(不要担心覆盖插入元素的值,反正前面使用临时变量存储起来了)
        for(j = i - 1;j >= high + 1;j--){
            L->data[j + 1] = L->data[j];
        }

        // 这里使用high + 1保险一点,因为for循环退出后的j会加1的
        L->data[high + 1] = temp;
    }
    return OK;
}

    // 顺序表的希尔排序
Status shellSort(SqList* L){
    int i,j,increment;
    ElemType temp;

    // 增量为表长的一半并不断折半
    for(increment = L->length / 2;increment > 0;increment /= 2){
        // 分组插入排序从增量位置向后开始
        for(i = increment;i < L->length;i++){
            // 保存当前组的插入元素
            temp = L->data[i];
            // 组内插入排序,增量不再是1而是指定值
            for(j = i;j >= increment;j -= increment){
                if(temp < L->data[j - increment]){
                    L->data[j] = L->data[j - increment];
                }else{
                    break;
                }
            }
            // 插入元素
            L->data[j] = temp;
        }
    }
    return OK;
}

        }
    }
    return OK;

    // L->next p->next->next->next->...
    p = L->next->next;

        }

        p = q;

#include <stdio.h>

#include <stdlib.h>

#include <stdbool.h> // 引入stdbool.h头文件就可以使用布尔类型变量

#include <time.h> // 生成随机数种子


// 使用const代替#define
#define MAXSIZE 20 // 存储空间初始分配量
#define OK 1
#define ERROR 0
#define TRUE 1
#define FALSE 0

typedef int ElemType; // 定义顺序表中元素数据类型
typedef int Status; // 表示状态的int,用于判断返回值是上面宏定义的1还是0
typedef struct{
    ElemType* base; // 栈底指针
    ElemType* top; // 栈顶指针
    int stacksize; // 栈可用最大容量
}SqStack;

Status initStack(SqStack &S){
    // 初始化栈底为栈数组容器
    // 进行初始化的指针变量默认值为0
    S.base = new ElemType[MAXSIZE];
    if(!S.base){
        return ERROR;
    }
    
    // 让栈顶等于栈底,这样栈顶就等于栈容器了.入栈就是向数组中插入元素,栈顶也上升
    S.top = S.base; // 初始化栈顶等于栈底,为空栈
    S.stacksize = MAXSIZE;
    return OK;
}

Status push(SqStack &S, ElemType e){
    if(S.top - S.base == S.stacksize){
        return ERROR;
    }

    // (*S.top++)就相当于 S.top[i++]
    // *a++就相当于a[i + 1]
    // 注意,先插入再自增,这样栈顶就永远为空(其实也可能是为了防止数据量过大栈数组索引越界的情况)
    *S.top++ = e;
    return OK; 
}

Status pop(SqStack &S,ElemType &e){
    if(S.top == S.base){
        return ERROR;
    }

    e = *--S.top;
    return OK;
}

ElemType getTopElem(SqStack S){
    if(S.top != S.base){
        return *(S.top-1);
    }
}

Status displayStack(SqStack S){
    // 如果栈为空,则不需要遍历
    if(S.top == S.base){
        return ERROR;
    }

    // 引用传递
    ElemType* p = S.base;
    while(p != S.top){
        printf("%d\t", *p);
        p++;
    }
    printf("\n");
    return OK;
}

int main(){
    SqStack S;
    int i;

    initStack(S);

    for(i = 0;i < 10;i++){
        push(S, rand() % 100 + 1);
    }
    displayStack(S);

    ElemType e;
    pop(S,e);
    printf("%d\n",e);

    printf("%d\n", getTopElem(S));

    return 0;
}

#include <stdio.h>

#include <stdlib.h>

#include <stdbool.h> // 引入stdbool.h头文件就可以使用布尔类型变量

#include <time.h> // 生成随机数种子


// 使用const代替#define
#define MAXSIZE 20 // 存储空间初始分配量
#define OK 1
#define ERROR 0
#define TRUE 1
#define FALSE 0

typedef int ElemType; // 定义顺序表中元素数据类型
typedef int Status; // 表示状态的int,用于判断返回值是上面宏定义的1还是0
typedef struct StackNode{
    ElemType data;
    struct StackNode* next;
}StackNode,*LinkStack;

Status initStack(LinkStack &S){
    S = NULL; // 就初始化一个栈顶
    return OK;
}

Status Push(LinkStack &S,ElemType e){
    // 这里用什么都行,但是由于p是要对S进行操作的,所以用LinkStack
    LinkStack p = new StackNode;
    p->data = e;

    // 新节点为栈顶
    p->next = S;

    // 为了下一次插入操作,让S再次等于栈顶
    S = p;
    return OK;
}

Status Pop(LinkStack &S,ElemType &e){
    if(S == NULL){
        return ERROR;
    }

    e = S->data;
    // 用一个临时变量保存栈顶元素
    LinkStack p = S;
    S = S->next;
    delete p;
    return OK;
}

ElemType GetTop(LinkStack &S){
    if(S != NULL){
        return S->data;
    }
}

void displayLinkStack(LinkStack &S){
    LinkStack p = S;
    while(p != NULL){
        printf("%d-->\t",p->data);
        p = p->next;
    }
    printf("\n");
}


int main(){
    LinkStack S;

    initStack(S);
    
    srand(unsigned(time(NULL)));
    int i;
    for(i = 0;i < 10;i++){
        Push(S,rand() % 100 + 1);
    }
    displayLinkStack(S);
    
    ElemType e;
    Pop(S,e);
    printf("%d\n",e);
    displayLinkStack(S);

    printf("%d\n",GetTop(S));

    return 0;
}

typedef int Status; // 表示状态的int,用于判断返回值是上面宏定义的1还是0

Status matchBracket(SqStack &S){
    int flag = 1; // 用于判断匹配是否成功,如果flag = 0证明匹配失败,直接退出程序
    ElemType e; 
    
    char input;

    printf("请输入内容(限制字符为'('、'{'、'['以及它们的互补括号):");
    scanf("%c", &input);

    while(input != '#' && flag){
        // 如果输入的是左括号,则入栈
        if((input == '(') || (input == '{') || (input == '[')){
            push(S,input);

        // 如果输入的是右括号,则:
            // 1.判断栈是否为空
            // 2.判断输入的括号是否和栈顶匹配
        }else if(input == ')'){
            if((S.top != S.base) && getTopElem(S) == '('){
                pop(S,e);
            }else{
                flag = 0;
                break;
            }
        }else if(input == '}'){
            if((S.top != S.base) && getTopElem(S) == '{'){
                pop(S,e);
            }else{
                flag = 0;
                break;
            }
        }else if(input == ']'){
            if((S.top != S.base) && getTopElem(S) == '['){
                pop(S,e);
            }else{
                flag = 0;
                break;
            }
        }

        /*
         * 第一次输入字符串之后回车，最后‘\n‘没有给input,而是保存在缓存中，当第二次需要输入时，直接把缓存中的字符赋值
         */
        while(getchar() != '\n'){
            continue;
        }
        printf("请输入内容(限制字符为'('、'{'、'['以及它们的互补括号):");
        scanf("%c", &input);
    }

    if((S.top == S.base) && flag){
        return OK;
    }else{
        return ERROR;
    }
}

Status initStack(SqStack &S){
    // 初始化栈底为栈数组容器
    // 进行初始化的指针变量默认值为0
    S.base = new ElemType[MAXSIZE];
    if(!S.base){
        return ERROR;
    }
    
    // 让栈顶等于栈底,这样栈顶就等于栈容器了.入栈就是向数组中插入元素,栈顶也上升
    S.top = S.base; // 初始化栈顶等于栈底,为空栈
    S.stacksize = MAXSIZE;
    return OK;
}

Status push(SqStack &S, ElemType e){
    if(S.top - S.base == S.stacksize){
        return ERROR;
    }

    // (*S.top++)就相当于 S.top[i++]
    // *a++就相当于a[i + 1]
    // 注意,先插入再自增,这样栈顶就永远为空(其实也可能是为了防止数据量过大栈数组索引越界的情况)
    *S.top++ = e;
    return OK; 
}

Status pop(SqStack &S,ElemType &e){
    if(S.top == S.base){
        return ERROR;
    }

    e = *--S.top;
    return OK;
}

ElemType getTopElem(SqStack S){
    if(S.top != S.base){
        return *(S.top-1);
    }
}

Status displayStack(SqStack S){
    // 如果栈为空,则不需要遍历
    if(S.top == S.base){
        return ERROR;
    }

    // 引用传递
    ElemType* p = S.base;
    while(p != S.top){
        printf("%d\t", *p);
        p++;
    }
    printf("\n");
    return OK;
}

    }else{
        printf("匹配失败!!!");
    }

    return 0;

#define MAXSIZE 20 // 存储空间初始分配量
#define OK 1
#define ERROR 0
#define TRUE 1
#define FALSE 0

typedef int ElemType; // 定义顺序表中元素数据类型
typedef int Status; // 表示状态的int,用于判断返回值是上面宏定义的1还是0
typedef struct{
    ElemType* base; // 存储空间的基地址
    int front; // 队头指针
    int rear; // 队尾指针
}SqQueue;

// base代表队列数组,front是对头,rear是队尾
Status initQueue(SqQueue &Q){
    Q.base = new ElemType[MAXSIZE];
    if(!Q.base){
        return ERROR;
    }

    // 当队头和队尾相等时候代表栈满
    /*
     * 1   --->   front
     * 2 
     * 3                                                                                 @重点:从front向上到rear是队列数据域
     * 
     * 5   --->   rear
     * 6
     * 7
     */
    Q.front = Q.rear; 
    return OK;
}

Status enQueue(SqQueue &Q,ElemType e){
    // 两种情况队列满(Q.rear + 1不能放置元素,就是为了队列循环而放置的):
    // 1.只进不出,rear一路+++到队列存储空间尾前一个元素,这时候索引+1除以队列空间长度就是等于front(队列首元素)
    // 2.先进后出,最后rear和front在队列中间两个元素,他们俩中间隔一个元素而不相等
    if((Q.rear + 1) % MAXSIZE == Q.front){
        return ERROR;
    }
    Q.base[Q.rear] = e;
    // 每增加一个队尾就向上增加一个,同时还要注意是否越界
    // 如果rear是队尾的前一个,那么它加上1就是队列最大长度的数值,除以MAXSIZE就等于0,达到既让队尾最后一个元素不放置元素又可以不发生队溢出达到循环的目的
    Q.rear = (Q.rear + 1) % MAXSIZE;
    return OK;
}

Status deQueue(SqQueue &Q,ElemType &e){
    if(Q.rear == Q.front){
        return ERROR;
    }
    e = Q.base[Q.rear];
    Q.front = (Q.front + 1) % MAXSIZE;
    return OK;
}

ElemType getTop(SqQueue Q){
    if(Q.rear != Q.front){
        return Q.base[Q.front];
    }
}

Status displayQueue(SqQueue Q){
    if(Q.front == Q.rear){
        return ERROR;
    }
    
    // 永远记得,Q.front "向上" 到Q.rear是队列元素的数据域
    while(Q.front != Q.rear){
        printf("%d\t", Q.base[Q.front]);
        Q.front = (Q.front + 1) % MAXSIZE;
    }

    return OK;
}

    initQueue(Q);

    int i;
    for(i = 0;i < 10;i++){
        enQueue(Q,i);
    }

    displayQueue(Q);

    ElemType e;
    deQueue(Q,e);
    printf("%d\n",e);
    displayQueue(Q);

    printf("%d\n",getTop(Q));

    return 0;
}

    ElemType data; // 节点数据域
    struct QNode* next; // 指针
}QNode,*QueuePtr; 
typedef struct{
    QueuePtr front;
    QueuePtr rear;
}LinkQueue; // 队列指针,队列头指针,队列尾指针


Status initQueue(LinkQueue &Q){
    // 构造一个空队列
    // 队列讲究队头和队尾的关系
    Q.front = Q.rear = new QNode;
    Q.front->next = NULL;
    return OK;
}


// 从enQueue和deQueue看出,Q.front是空的,出队操作也是头指针->next
Status enQueue(LinkQueue &Q, ElemType e){
    QNode* p = new QNode;
    p->data = e;

    // 将新节点插入队尾
    p->next = NULL; 
    Q.rear->next = p;

    // 队尾前进一步
    Q.rear = p;
    return OK;
}

Status deQueue(LinkQueue &Q,ElemType &e){
    if(Q.front == Q.rear){
        return ERROR;
    }
    QNode* p = new QNode;

    // 要出队的节点
    p = Q.front->next;
    e = p->data;

    // 删除节点
    Q.front->next = p->next;
    
    // 如果删除的节点正好是队尾,说明删除后队列空了,front = rear 
    if(Q.rear == p){
        Q.rear = Q.front;
    }
    delete p;
    return OK;
}

ElemType getTop(LinkQueue &Q){
    if(!(Q.rear == Q.front)){
        return Q.front->next->data;
    }
}

Status displayQueue(LinkQueue Q){
    QNode* p = Q.front;

    while(p){
        printf("%d\t", p->data);
        p = p->next;
    }
    printf("\n");
    return OK;
}

int BFMatch(String S,String T,int pos){
    // 主串从pos开始匹配,子串从1开始匹配
    int i = pos,j = 0;

    // 当其中有一个数组匹配完即退出
    while(i <= (S.length - 1) && j <= (T.length - 1)){
        if(S.ch[i] == T.ch[j]){
            ++i;
            ++j;
        }else{
            // i - j + 2等于上一次主串匹配位置的下一个
            i = (i - j + 2);
            j = 0;
        }
    }

    if(j > T.length - 1){
        // 此时i和j都等于他们自身字符串长度 + 1,(i - T.length)是匹配成功的那次->主串开始匹配位置的索引
        // 这里加上1纯属是为了返回匹配到的位置而不是索引数值
        return (i - T.length + 1); 
    }else{
        return 0;
    }
}

    scanf("%s",T.ch);

    T.length = strlen(T.ch);

    int res = BFMatch(S,T,0);

#include<stdio.h>

#include<stdlib.h>

typedef char ElemType; //ElemType类型定义
 
struct GNode //广义表结点类型定义
{
    int tag;   //标志域
    union
    {
        ElemType data;          //值域
        struct GNode* sublist;  //子表的表头指针域
    };
    struct GNode* next;      //指向后继结点的指针域
};
 
//1、建立广义表
//假定广义表的元素类型ElemType为字符类型char，每个单元素的值被限定为英文字母，
//输入格式为： (a,(#),b,(d,(e)));
//整个表最后以分号作为结束符，#表示空表
//建立广义表需要进行子表递归和后继表递归
void CreateGList_char(struct GNode** GL)
{
    char ch;
    scanf("%c", &ch);
    if (ch == '#') //若输入为空格，则置表头指针为空
        *GL = NULL;
    else if (ch == '(') //若输入为左括号则建立由*GL所指向的子表结点并递归构造子表
    {
        *GL = malloc(sizeof(struct GNode));
        (*GL)->tag = 1;
        CreateGList_char(&((*GL)->sublist)); //递归
    }
    else //若输入为字符则建立由*GL所指向的单元素结点
    {
        *GL = malloc(sizeof(struct GNode));
        (*GL)->tag = 0;
        (*GL)->data = ch;
    }
    scanf("%c", &ch); //此处读入的字符必为逗号、括号或分号
    if (*GL == NULL); //若*GL为空，什么都不做
    else if (ch == ',')//若输入为逗号则递归构造后续表
        CreateGList_char(&((*GL)->next)); //递归
    else if (ch == ')' || ch == ';') //若输入为右括号或分号则置*GL的后继指针域为空
        (*GL)->next = NULL;
}
 
//2、打印输出广义表
//打印输出上面的广义表同样需要递归子表和递归后继表
void PrintfGList_char(struct GNode* GL)
{
    if (GL->tag == 1) //表结点的处理情况
    {
        printf("%c", '('); //先输出左括号，作为开始符号
        if (GL->sublist == NULL) //若子表为空
            printf("%c", '#');
        else
            PrintfGList_char(GL->sublist); //递归
        printf("%c", ')'); //当一个子表输出结束后，输出一个右括号终止符
    }
    else //单元素结点的处理情况
        printf("%c", GL->data);
    if (GL->next != NULL) //若结点还有后继表，则递归输出
    {
        printf("%c", ','); //先输出逗号分隔符
        PrintfGList_char(GL->next); //递归
    }
}
 
//3、求广义表的长度
//广义表中，同一层次的每个结点是通过next域链接起来的，所以可把它看做是由next域链接起来的单链表。
//这样求广义表的长度就是求单链表的长度
//如单链表非空，其长度等于1加其后继单链表的长度，若单链表为空，则长度为0，这是递归的终止条件
int LenthGList(struct GNode* GL)
{
    if (GL != NULL)
        return 1 + LenthGList(GL->next);
    else
        return 0;
}
 
//4、求广义表的深度
//广义表深度的递归定义是它等于所有子表中的最大深度加1，若一个表为空或仅由单元素组成，则深度为1
int DepthGList(struct GNode* GL)
{
    int max = 0; //所有子表中最大深度，初始值为0
    while (GL != NULL)
    {
        if (GL->tag == 1)
        {
            int dep = DepthGList(GL->sublist); //递归调用求出一个子表的深度，用dep表示任一子表的深度
            if (dep > max)
                max = dep; //让max始终为同一层所求过的子表中深度的最大值
        }
        GL = GL->next;     //使GL指向同一层的下一个结点
    }
    return max + 1; //返回表的深度
}
 
//主函数
void main()
{
    struct GNode* gl;
    printf("输入一个广义表，输入格式为： (a,(#),b,(d,(e))); \n");
    CreateGList_char(&gl);
    printf("输出广义表：\n");
    PrintfGList_char(gl);
    printf("\n");
    printf("广义表的长度为：");
    printf("%d\n", LenthGList(gl->sublist));
    printf("广义表的深度为：");
    printf("%d\n", DepthGList(gl->sublist));
}

树的相关概念

  • 结点的度

    结点拥有的子树个数成为节点的度

  • 树的度

    树中各节点度的最大值

  • 叶子节点/终端节点

    度为0的节点,称为叶子节点或终端节点

  • 分支节点/非终端节点

    度不为0的节点称为分支节点或非终端节点

  • 双亲和孩子

    节点的子树的根节点称为该节点的孩子

    对应的,该节点称为孩子的双亲

  • 兄弟

    同一个节点的孩子之间互称兄弟

  • 祖先

    从该节点到根节点所经过的全部节点,统统都是该节点的祖先

  • 子孙

    以某一结点为根节点的子树,子树中全部节点都称为根节点的子孙

  • 层次

    从根节点开始算为第一层

  • 堂兄弟

    双亲在同一层的互为堂兄弟

  • 树的深度或高度

    树中节点的最大层数称为树的深度或高度

  • 有序树

    若将树中各个节点的各子树都看成是从左到右有序的,则称该树为有序树

  • 森林

    是n颗互不相交的树的集合

    对于树中每个节点而言,其子树的集合就是森林

二叉树

二叉树的性质

  • 二叉树的第i层上至多有2^(i-1)个节点

  • 深度为k的二叉树至多有2^(k)-1个节点

  • 对任意一颗二叉树,如果其终端节点数(度为0的节点)为n0,度为2的节点数为n2,则n0 = n2 + 1

    • 证明

      n为树节点总数,n0、n1、n2是度为0、度为1、度为2的节点

      二叉树中所有节点无非就是度为0、度为1、度为2

      所以就有 n = n0 + n1 + n2 ①

      再看二叉树中的分支数,除了根节点外,每棵树都有一个分支进入

      设B为分支总数,则 n = B + 1 ②

      由于这些分支是由度为1或度为2的节点射出的,所以B = n1 + 2n2 ③

      可得n = n1 + 2n2 + 1 ④

      联立①和④

      n0 = n2 + 1

二叉树的遍历

先序遍历(中左右)

先序遍历可以想象为，一个小人从一棵二叉树根节点为起点，沿着二叉树外沿，逆时针走一圈回到根节点，路上遇到的元素顺序，就是先序遍历的结果

  • 先序遍历结果为：A B D H I E J C F K G

  • 动画演示

记住小人沿着外围跑一圈（直到跑回根节点），多看几次动图便能理解

中序遍历(左中右)

中序遍历可以看成，二叉树每个节点，垂直方向投影下来（可以理解为每个节点从最左边开始垂直掉到地上），然后从左往右数，得出的结果便是中序遍历的结果

  • 中序遍历结果为:H D I B E J A F K C G

  • 动画展示

记住，中序遍历就是从最左边开始，把每个节点垂直投影到同一直线上，然后从左往右读值就可以了，多看几遍动图就理解了

后序遍历(左右中)

后序遍历就像是剪葡萄，我们要把一串葡萄剪成一颗一颗的。

还记得我上面提到先序遍历绕圈的路线么？（不记得翻上面理解）

就是围着树的外围绕一圈，如果发现一剪刀就能剪下的葡萄（必须是一颗葡萄）（也就是葡萄要一个一个掉下来，不能一口气掉超过1个这样），就把它剪下来，组成的就是后序遍历了。

后序遍历中，根节点默认最后面

  • 后序遍历结果：H I D J E B K F G C A

层次遍历

        return;

    postOrder(T->rchild);
